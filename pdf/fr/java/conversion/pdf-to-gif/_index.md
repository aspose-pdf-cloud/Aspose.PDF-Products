```java

    package com.aspose.asposecloudpdfusecases.conversions;

    import java.io.File;
    import java.nio.file.Path;
    import java.nio.file.StandardCopyOption;
    import com.aspose.asposecloudpdf.api.PdfApi;

    public class ConvertPdfToGif {
        public static void Convert() {
            String REMOTE_FOLDER   = "Your_Temp_Pdf_Cloud";
            String LOCAL_FOLDER    = "c:\\Samples";
            String PDF_FILE_NAME  = "sample.pdf";
            String GIF_OUTPUT = "convert_pdf_gif_output.gif";
            Integer PAGE_NUMBER = 2;

            try {
                PdfApi pdfApi = new PdfApi(API_KEY, API_SECRET);

                // upload local PDF file to remote storage
                File file = new File(Path.of(LOCAL_FOLDER, PDF_FILE_NAME).toString());
                String srcPath = Path.of(REMOTE_FOLDER, PDF_FILE_NAME).toString();
                pdfApi.uploadFile(srcPath, file, null);
                System.out.println(String.format("Files %s successfully uploaded!", PDF_FILE_NAME));

            // convert PDF to GIF and download image locally
                File f = pdfApi.getPageConvertToGif(PDF_FILE_NAME, PAGE_NUMBER, null, null, REMOTE_FOLDER, null, null);
                java.nio.file.Files.copy(Path.of(f.getPath()), Path.of(LOCAL_FOLDER, GIF_OUTPUT), StandardCopyOption.REPLACE_EXISTING);
                System.out.println(String.format("File '%s' successfully dowloaded!", Path.of(LOCAL_FOLDER, GIF_OUTPUT).toString()));
                System.out.println("Successfully converted PDF document page to GIF ! '" + GIF_OUTPUT + "'");
            } catch (Exception e) {
                e.printStackTrace();
            }
        }
    }
```
