```python

    from asposepdfcloud import PdfApi, ApiClient
    import shutil
    import os
    import json
    from pathlib import Path
    import logging

    # Configure logging
    logging.basicConfig(level=logging.INFO, format="%(asctime)s - %(levelname)s - %(message)s")

    class PdfHtmlToPdfConvert:
        """Class for converting to PDF using Aspose PDF Cloud API."""
        def convert(self):
            """Convert Html to PDF document."""
            localFolder = "C:\Samples"
            storageHtmlName = "sample.html"
            storageTempFolder = "TempPdfCloud"
            outputFileName = "output_convert_html.pdf"
            htmlZipFileName  = "sample_html.zip"

            # Get your AppSid and AppSecret from https://dashboard.aspose.cloud (free registration required). 
            self.pdf_api = PdfApi(ApiClient(AppSecret, AppSid))

            if self.pdf_api:
                file_path = localFolder + "/" + htmlZipFileName
                self.pdf_api.upload_file(storageTempFolder + '/' + htmlZipFileName, file_path)
                
                opts = { 
                    "dst_folder": storageTempFolder,
                    "height" : 650,
                    "width" : 250,
                    "html_file_name": storageHtmlName
                }

                try:
                    response = self.pdf_api.put_html_in_storage_to_pdf(outputFileName, storageTempFolder + '/' + htmlZipFileName, **opts)
                    if response.code == 200:
                        temp_file = self.pdf_api.download_file(storageTempFolder + '/' + outputFileName)
                        local_path = localFolder + '/' + outputFileName
                        shutil.move(temp_file, local_path)
                        logging.info(f"html_convert(): HTML file successfully converted to PDF: '{outputFileName}'.")
                    else:
                        logging.error(f"html_convert(): Failed to convert HTML file to PDF. Response code: {response.code}")
                except Exception as e:
                    logging.error(f"html_convert(): Error while converting HTML file: {e}")

```
