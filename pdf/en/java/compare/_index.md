```java

    package com.aspose.asposecloudpdfusecases.compares;

    import java.io.File;
    import java.nio.file.Path;
    import java.nio.file.StandardCopyOption;
    import com.aspose.asposecloudpdf.api.PdfApi;
    import com.aspose.asposecloudpdf.model.AsposeResponse;
    import com.aspose.asposecloudpdfusecases.AppHelper;

    public class ComparePdfDocuments {
        public static void Compare() {
            String LOCAL_FOLDER = "C:\\Samples\\";
            String REMOTE_FOLDER = "Your_Temp_Pdf_Cloud";
            String PDF_DOCUMENT_1 = "sample_compare_1.pdf";
            String PDF_DOCUMENT_2 = "sample_compare_2.pdf";
            String PDF_OUTPUT = "output_compare.pdf";

            try{
                PdfApi pdfApi = new PdfApi(API_KEY, API_SECRET);
                
                // upload local PDF file to remote storage
                File file = new File(Path.of(LOCAL_FOLDER, PDF_DOCUMENT_1).toString());
                pdfApi.uploadFile(Path.of(REMOTE_FOLDER , PDF_DOCUMENT_1).toString(), file, null);
                System.out.println(String.format("File '%s' successfully uploaded!", Path.of(LOCAL_FOLDER, PDF_DOCUMENT_1).toString()));

                file = new File(Path.of(LOCAL_FOLDER, PDF_DOCUMENT_2).toString());
                pdfApi.uploadFile(Path.of(REMOTE_FOLDER , PDF_DOCUMENT_2).toString(), file, null);
                System.out.println(String.format("File '%s' successfully uploaded!", Path.of(LOCAL_FOLDER, PDF_DOCUMENT_2).toString()));

                String remotePdfOutput = Path.of(REMOTE_FOLDER , PDF_OUTPUT).toString();
                AsposeResponse response = pdfApi.postComparePdf(
                    Path.of(REMOTE_FOLDER , PDF_DOCUMENT_1).toString(),
                    Path.of(REMOTE_FOLDER , PDF_DOCUMENT_2).toString(),
                    remotePdfOutput,
                    null);            

                if (response.getCode() != 200)
                    System.err.println("Error: unexpected error when comparing documents! '" + response.getStatus() + "'");
                else {
                    // download changed PDF file from remote folder...
                        File f = pdfApi.downloadFile(remotePdfOutput, null, null);
                        java.nio.file.Files.copy(Path.of(f.getPath()), Path.of(LOCAL_FOLDER, PDF_OUTPUT), StandardCopyOption.REPLACE_EXISTING);
                        System.out.println(String.format("File '%s' successfully dowloaded!", Path.of(LOCAL_FOLDER, PDF_OUTPUT).toString()));                
                        System.out.println("Successfully compared documents ! '" + PDF_OUTPUT + "'");
                }

            } catch (Exception e) {
                e.printStackTrace();
            }
        }
    }
    
```
