```java

    package com.aspose.asposecloudpdfusecases.images;

    import java.io.File;
    import java.nio.file.Path;
    import java.nio.file.StandardCopyOption;
    import com.aspose.asposecloudpdf.api.PdfApi;
    import com.aspose.asposecloudpdf.model.ImageResponse;
    import com.aspose.asposecloudpdfusecases.AppHelper;

    public class ImagesGetById {
        public static void Extract() {
            String LOCAL_FOLDER         = "C:\\Samples\\";
            String REMOTE_FOLDER        = "Your_Temp_Pdf_Cloud";
            String PDF_DOCUMENT         = "sample.pdf";
            String IMAGE_ID             = "IMAGE_ID_VALUE";
            String IMAGE_OUTPUT_FILE    = "output_images_extract.jpg";
            Integer IMAGE_OUTPUT_WIDTH  = 300;
            Integer IMAGE_OUTPUT_HEIGHT = 300;

            try{
                PdfApi pdfApi = new PdfApi(API_KEY, API_SECRET);

                // upload local PDF file to remote storage
                File file = new File(Path.of(LOCAL_FOLDER, PDF_DOCUMENT).toString());
                pdfApi.uploadFile(Path.of(REMOTE_FOLDER, PDF_DOCUMENT).toString(), file, null);
                System.out.println(String.format("File '%s' successfully uploaded!", Path.of(LOCAL_FOLDER, PDF_DOCUMENT).toString()));

                ImageResponse response = pdfApi.getImage(PDF_DOCUMENT, IMAGE_ID, null, REMOTE_FOLDER);
                if (response.getCode() != 200)
                    System.err.println("Error: unexpected error when extrcting image from document! '" + response.getStatus() + "'");
                else {
                    File imageFile = pdfApi.getImageExtractAsJpeg(PDF_DOCUMENT, IMAGE_ID, IMAGE_OUTPUT_WIDTH, IMAGE_OUTPUT_HEIGHT, null, REMOTE_FOLDER);
                    java.nio.file.Files.copy(Path.of(imageFile.getPath()), Path.of(LOCAL_FOLDER, IMAGE_OUTPUT_FILE), StandardCopyOption.REPLACE_EXISTING);
                    System.out.println(String.format("File '%s' successfully dowloaded!", Path.of(LOCAL_FOLDER, IMAGE_OUTPUT_FILE).toString()));                
                    System.out.println("Successfully extracted image from document !\n" + IMAGE_OUTPUT_FILE + "\n");
                }
            } catch (Exception e) {
                e.printStackTrace();
            }
        }
    }

```
