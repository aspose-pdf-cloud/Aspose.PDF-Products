        """ Upload a local fileName to the Aspose Cloud server. """
        file_path = localFolder / documentName
        self.pdf_api.upload_file(os.path.join(remoteFolder, fileName), file_path)

        field = Field(
            name=fieldName,
            type=FieldType.TEXT,
