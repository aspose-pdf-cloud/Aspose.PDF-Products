```go

    package main

    import (
        "fmt"

        asposepdfcloud "github.com/aspose-pdf-cloud/aspose-pdf-cloud-go/v25"
    )

    func ExtractPageInfo(pdf_api *asposepdfcloud.PdfApiService, document string, pageNumber int32, remoteFolder string) {
        // Get page information of the PDF document.
        uploadFile(pdf_api, document)

        args := map[string]interface{}{
            "folder": remoteFolder,
        }

        result, httpResponse, err := pdf_api.GetPage(document, pageNumber, args)
        if err != nil {
            fmt.Println(err.Error())
        } else if httpResponse.StatusCode < 200 || httpResponse.StatusCode > 299 {
            fmt.Println("Unexpected error!")
        } else {
            fmt.Println("Successfully extract page :", pageNumber, "Width :", result.Page.Rectangle.URX, "Height: ", result.Page.Rectangle.URY)

        }
    }
```

Mengekstrak halaman dari PDF adalah tugas umum yang melayani berbagai tujuan di berbagai konteks. Proses ini melibatkan pemilihan halaman tertentu dari dokumen yang lebih besar untuk membuat file PDF baru yang terpisah. Memahami alasan di balik praktik ini dapat membantu dalam mengelola dan memanfaatkan dokumen PDF secara efektif. File PDF yang besar bisa jadi sulit untuk dibagikan atau disimpan. Dengan mengekstrak hanya halaman yang diperlukan, pengguna dapat membuat file yang lebih kecil dan lebih mudah dikelola. Ini sangat berguna ketika hanya sebagian dari dokumen yang relevan untuk tujuan tertentu. Misalnya, menghapus halaman yang tidak perlu dapat secara signifikan mengurangi ukuran file, membuatnya lebih mudah untuk ditangani dan didistribusikan.
Mengekstrak halaman memungkinkan pengguna untuk menggunakan kembali konten untuk aplikasi yang berbeda. Misalnya, seseorang dapat mengekstrak halaman dari laporan komprehensif untuk membuat ringkasan mandiri atau mengisolasi data tertentu untuk analisis. Ini memungkinkan penggunaan kembali konten yang ada tanpa perlu membuat ulang informasi dari awal.
